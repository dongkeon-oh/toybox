CREATE TABLE TB_COMMON_GROUP(
    CGR_GROUP	    	VARCHAR(16) PRIMARY KEY,
    CGR_GROUP_NAME	VARCHAR(100) NOT NULL,
    CGR_NOTE		VARCHAR(1000),
    CGR_USEYN		CHAR(1) DEFAULT 'Y' NOT NULL
);


CREATE TABLE TB_COMMON_CODE(
    CCD_CODE	    	VARCHAR(100) PRIMARY KEY,
    CCD_GROUP		VARCHAR(16) REFERENCES TB_COMMON_GROUP(CGR_GROUP) NOT NULL,
    CCD_CODENAME		VARCHAR(100) NOT NULL,
    CCD_DETAIL1		VARCHAR(1000),
    CCD_DETAIL2		VARCHAR(1000),
    CCD_DETAIL3		VARCHAR(1000),
    CCD_NOTE		VARCHAR(1000),
    CCD_ORDER		NUMBER(5)
);

CREATE TABLE TB_USER(
    USR_ID			VARCHAR(20) PRIMARY KEY,
    USR_PASSWORD		VARCHAR(256) NOT NULL,
    USR_NAME		VARCHAR(10) NOT NULL,
    USR_IMAGE		VARCHAR(100)    DEFAULT 'user_default.jpg' NOT NULL,
    USR_EMAIL 		VARCHAR(100) NOT NULL,
    USR_SMS		VARCHAR(13),
    USR_KAKAO		VARCHAR(100),
    USR_ACTIVE		VARCHAR2(100) REFERENCES TB_COMMON_CODE(CCD_CODE) NOT NULL,
    USR_QUESTION		VARCHAR2(100) REFERENCES TB_COMMON_CODE(CCD_CODE) NOT NULL,
    USR_ANSWER		VARCHAR(100) NOT NULL,
    USR_TYPE     		VARCHAR2(100) REFERENCES TB_COMMON_CODE(CCD_CODE) NOT NULL,  
);

CREATE TABLE TB_ITEM(
    ITM_ID			VARCHAR2(8) PRIMARY KEY,
    ITM_NAME		VARCHAR2(100) NOT NULL,
    ITM_TYPE		VARCHAR2(100) REFERENCES TB_COMMON_CODE(CCD_CODE) NOT NULL, 
    ITM_OWNER		VARCHAR2(20) REFERENCES TB_USER(USR_ID) NOT NULL,
    ITM_MAINITEM		VARCHAR2(8) DEFAULT 'NO_ITEM' NOT NULL,
    ITM_NOTE		VARCHAR2(1000),
    ITM_IMAGE		VARCHAR2(100) DEFAULT 'item_default.jpg' NOT NULL,
    ITM_USEYN       		VARCHAR2(100) REFERENCES TB_COMMON_CODE(CCD_CODE) NOT NULL
);

CREATE TABLE TB_CONDITION(
    CDT_ID			VARCHAR(16) PRIMARY KEY,
    CDT_CONDITION		VARCHAR(8) NOT NULL,
    CDT_ITEM		VARCHAR(8) REFERENCES TB_ITEM(ITM_ID) NOT NULL,
    CDT_USER		VARCHAR(20) REFERENCES TB_USER(USR_ID) NOT NULL,
    CDT_LOCATION		VARCHAR(20) REFERENCES TB_USER(USR_ID) NOT NULL,
    CDT_RETURN		VARCHAR(8) NOT NULL,
    CDT_NOTE		VARCHAR(1000),
    CDT_TODATE		DATE NOT NULL,
    CDT_FROMDATE		DATE NOT NULL,
    CDT_RETURNDATE	DATE NOT NULL,
    CDT_CREATEDATE	DATE NOT NULL,
    CDT_PROCESS      	VARCHAR2(8) NOT NULL
);

CREATE SEQUENCE SEQ_ITEM_ID
START WITH 0
INCREMENT BY 1
MAXVALUE 9999
MINVALUE 0
;

CREATE SEQUENCE SEQ_CONDITION_PROCESS
START WITH 0
INCREMENT BY 1
MAXVALUE 99999999
MINVALUE 0
;

CREATE SEQUENCE SEQ_COMMON_ID
START WITH 0
INCREMENT BY 1
MAXVALUE 99999999
MINVALUE 0
;
